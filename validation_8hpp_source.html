46  TIME_SHIELD_CONSTEXPR const bool is_leap_year_ts(const ts_t& ts) {
47  // 9223372029693630000 - значение на момент 292277024400 от 2000 года
48  // Такое значение приводит к неправильному вычислению умножения n_400_years * SEC_PER_400_YEARS
49  // Поэтому пришлось снизить до 9223371890843040000
50  const int64_t BIAS_292277022000 = 9223371890843040000LL;
51  const int64_t BIAS_2000 = 946684800LL;
52 
53  int64_t y = MAX_YEAR;
54  int64_t secs = -((ts - BIAS_2000) - BIAS_292277022000);
55 
56  const int64_t n_400_years = secs / SEC_PER_400_YEARS;
57  secs -= n_400_years * SEC_PER_400_YEARS;

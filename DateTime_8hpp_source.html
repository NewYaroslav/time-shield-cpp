476 static bool try_parse_iso8601_buffer(const char* data, std::size_t size, DateTime& out) noexcept {
477 if (data == nullptr) {
478 return false;
479 }
480 DateTimeStruct dt = create_date_time_struct(0);
481 TimeZoneStruct tz = create_time_zone_struct(0, 0, true);
482 if (!time_shield::parse_iso8601(data, size, dt, tz)) {
483 return false;
484 }
485 const tz_t offset = time_zone_struct_to_offset(tz);
486 const ts_ms_t utc_ms = dt_to_timestamp_ms(dt) - offset_to_ms(offset);
487 out = from_unix_ms(utc_ms, offset);
488 return true;
489 }
490
491 static DateTime parse_iso8601_buffer(const char* data, std::size_t size) {
492 DateTime result;
493 if (!try_parse_iso8601_buffer(data, size, result)) {
494 throw std::invalid_argument("Invalid ISO8601 datetime");
495 }
496 return result;
497 }

74 template<class T1 = Weekday, class T2 = year_t, class T3 = int, class T4 = int>
75 TIME_SHIELD_CONSTEXPR inline T1 day_of_week_date(T2 year, T3 month, T4 day) {
76 year_t a = 0;
77 year_t y = 0;
78 year_t m = 0;
79 year_t R = 0;
80 a = (14 - month) / MONTHS_PER_YEAR;
81 y = year - a;
82 m = month + MONTHS_PER_YEAR * a - 2;
83 R = 7000 + ( day + y + (y / 4) - (y / 100) + (y / 400) + (31 * m) / MONTHS_PER_YEAR);
84 return static_cast<T1>(R % DAYS_PER_WEEK);
85 }

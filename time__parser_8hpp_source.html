153 const bool parse_time_zone(const std::string& tz_str, TimeZoneStruct &tz) {
154 if (tz_str.empty()) {
155 tz.hour = 0;
156 tz.min = 0;
157 tz.is_positive = true;
158 return true;
159 }
160 if (tz_str == "Z") {
161 tz.hour = 0;
162 tz.min = 0;
163 tz.is_positive = true;
164 return true;
165 }
166 tz.is_positive = (tz_str[0] == '+');
167 tz.hour = std::stoi(tz_str.substr(1, 2));
168 tz.min = std::stoi(tz_str.substr(4, 2));
169 return is_valid_time_zone(tz);

187 const bool parse_iso8601(
188 const std::string& input,
189 DateTimeStruct &dt,
190 TimeZoneStruct &tz) {
191 // Регулярное выражение для даты в формате ISO8601
192 static const std::regex date_regex(R"((\d{4})[-\/\.](\d{2})[-\/\.](\d{2}))");
193 std::smatch date_match;
194
195 // Регулярное выражение для времени в формате ISO8601 с часовым поясом и без
196 static const std::regex time_regex(R"((\d{2}):(\d{2}):(\d{2})(?:\.(\d+))?(Z|[+-]\d{2}:\d{2})?)");
197 std::smatch time_match;
198
199 dt = create_date_time_struct(0);
200 tz = create_time_zone_struct(0, 0);
201
202 // Парсинг даты
203 if (std::regex_search(input, date_match, date_regex)) {
204 dt.year = std::stoll(date_match[1].str());
205 dt.mon = std::stoi(date_match[2].str());
206 dt.day = std::stoi(date_match[3].str());
207 } else {
208 return false;
209 }
210
211 // Парсинг времени и часового пояса
212 if (std::regex_search(input, time_match, time_regex)) {
213 dt.hour = std::stoi(time_match[1].str());
214 dt.min = std::stoi(time_match[2].str());
215 dt.sec = std::stoi(time_match[3].str());
216 if (time_match[4].matched) {
217 dt.ms = std::stoi(time_match[4].str());
218 }
219 if (time_match[5].matched) {
220 if (!parse_time_zone(time_match[5].str(), tz)) return false;
221 }
222 return is_valid_date_time(dt);
223 }
224 return true;
225 }

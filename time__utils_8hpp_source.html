53 inline int64_t now_realtime_us() {
54# if defined(_WIN32)
55 static std::once_flag init_flag;
56 static int64_t s_perf_freq = 0;
57 static int64_t s_anchor_perf = 0;
58 static int64_t s_anchor_realtime_us = 0;
59
60 std::call_once(init_flag, [] {
61 LARGE_INTEGER freq, counter;
62 QueryPerformanceFrequency(&freq);
63 QueryPerformanceCounter(&counter);
64
65 s_perf_freq = freq.QuadPart;
66 s_anchor_perf = counter.QuadPart;
67
68 FILETIME ft;
69 GetSystemTimeAsFileTime(&ft);
70 ULARGE_INTEGER uli;
71 uli.LowPart = ft.dwLowDateTime;
72 uli.HighPart = ft.dwHighDateTime;
73 // Convert from 100ns since 1601 to microseconds since 1970
74 s_anchor_realtime_us = (uli.QuadPart - 116444736000000000ULL) / 10;
75 });
76
77 LARGE_INTEGER now;
78 QueryPerformanceCounter(&now);
79
80 int64_t delta_ticks = now.QuadPart - s_anchor_perf;
81 int64_t delta_us = (delta_ticks * 1000000) / s_perf_freq;
82
83 return s_anchor_realtime_us + delta_us;
84# else
85# error "now_realtime_us() is only supported on Windows."
86# endif

195 inline double get_cpu_time() noexcept {
196# if defined(_WIN32)
197 FILETIME create_time{}, exit_time{}, kernel_time{}, user_time{};
198 if (GetProcessTimes(GetCurrentProcess(), &create_time, &exit_time, &kernel_time, &user_time)) {
199 ULARGE_INTEGER li{};
200 li.LowPart = user_time.dwLowDateTime;
201 li.HighPart = user_time.dwHighDateTime;
202 return static_cast<double>(li.QuadPart) / 10000000.0;
203 }
204# elif defined(__unix__) || defined(__unix) || defined(unix) || (defined(__APPLE__) && defined(__MACH__))
205 // AIX, BSD, Cygwin, HP-UX, Linux, OSX, and Solaris
206# if defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0)
207 clockid_t id = (clockid_t)-1;
208# if defined(_POSIX_CPUTIME) && (_POSIX_CPUTIME > 0)
209 if (clock_getcpuclockid(0, &id) != 0) {
210# if defined(CLOCK_PROCESS_CPUTIME_ID)
211 id = CLOCK_PROCESS_CPUTIME_ID;
212# elif defined(CLOCK_VIRTUAL)
213 id = CLOCK_VIRTUAL;
214# endif
215 }
216# elif defined(CLOCK_PROCESS_CPUTIME_ID)
217 id = CLOCK_PROCESS_CPUTIME_ID;
218# elif defined(CLOCK_VIRTUAL)
219 id = CLOCK_VIRTUAL;
220# endif
221 if (id != (clockid_t)-1) {
222 struct timespec ts;
223 if (clock_gettime(id, &ts) == 0) {
224 return static_cast<double>(ts.tv_sec) + static_cast<double>(ts.tv_nsec) / 1e9;
225 }
226 }
227# endif
228
229# if defined(RUSAGE_SELF)
230 struct rusage usage{};
231 if (getrusage(RUSAGE_SELF, &usage) == 0) {
232 return static_cast<double>(usage.ru_utime.tv_sec) + static_cast<double>(usage.ru_utime.tv_usec) / 1e6;
233 }
234# endif
235
236# if defined(_SC_CLK_TCK)
237 struct tms t{};
238 if (times(&t) != (clock_t)-1) {
239 return static_cast<double>(t.tms_utime) / static_cast<double>(sysconf(_SC_CLK_TCK));
240 }
241# endif
242
243# if defined(CLOCKS_PER_SEC)
244 clock_t cl = clock();
245 if (cl != (clock_t)-1) {
246 return static_cast<double>(cl) / static_cast<double>(CLOCKS_PER_SEC);
247 }
248# endif
249# else
250# warning "get_cpu_time() may not work correctly: unsupported platform"

362 years -= n_400_years * 400;

365 secs += n_100_years * SEC_PER_100_YEARS;
366 years -= n_100_years * 100;
367
368 const int64_t n_4_years = years / 4;
369 secs += n_4_years * SEC_PER_4_YEARS;
370 years -= n_4_years * 4;
371
372 secs += years * SEC_PER_YEAR;

375 // Такое значение приводит к неправильному вычислению умножения n_400_years * SEC_PER_400_YEARS

378 constexpr int64_t BIAS_2000 = 946684800LL;

380 secs = BIAS_292277022000 - secs;
381 secs += BIAS_2000;

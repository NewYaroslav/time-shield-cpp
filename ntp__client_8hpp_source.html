152 struct ntp_packet {
153 uint8_t li_vn_mode; // Eight bits. li, vn, and mode.
154 // li. Two bits. Leap indicator.
155 // vn. Three bits. Version number of the protocol.
156 // mode. Three bits. Client will pick mode 3 for client.
157 uint8_t stratum; // Eight bits. Stratum level of the local clock.
158 uint8_t poll; // Eight bits. Maximum interval between successive messages.
159 uint8_t precision; // Eight bits. Precision of the local clock.
160 uint32_t root_delay; // 32 bits. Total round trip delay time.
161 uint32_t root_dispersion; // 32 bits. Max error aloud from primary clock source.
162 uint32_t ref_id; // 32 bits. Reference clock identifier.
163 uint32_t ref_ts_sec; // 32 bits. Reference time-stamp seconds.
164 uint32_t ref_ts_frac; // 32 bits. Reference time-stamp fraction of a second.
165 uint32_t orig_ts_sec; // 32 bits. Originate time-stamp seconds.
166 uint32_t orig_ts_frac; // 32 bits. Originate time-stamp fraction of a second.
167 uint32_t recv_ts_sec; // 32 bits. Received time-stamp seconds.
168 uint32_t recv_ts_frac; // 32 bits. Received time-stamp fraction of a second.
169 uint32_t tx_ts_sec; // 32 bits and the most important field the client cares about. Transmit time-stamp seconds.
170 uint32_t tx_ts_frac; // 32 bits. Transmit time-stamp fraction of a second.
171 };

180 void fill_packet(ntp_packet& pkt) const {
181 std::memset(&pkt, 0, sizeof(pkt));
182 pkt.li_vn_mode = (0 << 6) | (3 << 3) | 3; // LI=0, VN=3, Mode=3 (client)
183
184 const uint64_t now_us = time_shield::now_realtime_us();
185 const uint64_t sec = now_us / 1000000 + NTP_TIMESTAMP_DELTA;
186 const uint64_t frac = ((now_us % 1000000) * 0x100000000ULL) / 1000000;
187
188 pkt.tx_ts_sec = htonl(static_cast<uint32_t>(sec));
189 pkt.tx_ts_frac = htonl(static_cast<uint32_t>(frac));
190 }

193 bool parse_packet(const ntp_packet& pkt, int64_t& result_offset_us) const {
194 const uint64_t arrival_us = time_shield::now_realtime_us();
195
196 const uint64_t originate_us = ((static_cast<uint64_t>(ntohl(pkt.orig_ts_sec)) - NTP_TIMESTAMP_DELTA) * 1000000) +
197 (static_cast<uint64_t>(ntohl(pkt.orig_ts_frac)) * 1000000 / 0xFFFFFFFFull);
198 const uint64_t receive_us = ((static_cast<uint64_t>(ntohl(pkt.recv_ts_sec)) - NTP_TIMESTAMP_DELTA) * 1000000) +
199 (static_cast<uint64_t>(ntohl(pkt.recv_ts_frac)) * 1000000 / 0xFFFFFFFFull);
200 const uint64_t transmit_us = ((static_cast<uint64_t>(ntohl(pkt.tx_ts_sec)) - NTP_TIMESTAMP_DELTA) * 1000000) +
201 (static_cast<uint64_t>(ntohl(pkt.tx_ts_frac)) * 1000000 / 0xFFFFFFFFull);
202
203 // RFC 5905
204 result_offset_us = ((static_cast<int64_t>(receive_us) - static_cast<int64_t>(originate_us))
205 + (static_cast<int64_t>(transmit_us) - static_cast<int64_t>(arrival_us))) / 2;
206 return true;
207 }

51 size_t repeat_count,
52 ts_t ts,
53 tz_t utc_offset,
54 const DateTimeStruct& dt,
55 std::string& result) {
56 switch (last_char) {
57 case 'a':
58 if (repeat_count > 1) break;
59 result += to_str(day_of_week(dt.year, dt.mon, dt.day), FormatType::SHORT_NAME);
60 break;
61 case 'A':
62 if (repeat_count > 1) break;
63 result += to_str(day_of_week(dt.year, dt.mon, dt.day), FormatType::FULL_NAME);
64 break;
65 case 'I':
66 if (repeat_count == 1) {
67 char buffer[4] = {0};
68 snprintf(buffer, sizeof(buffer), "%.2d", hour24_to_12(dt.hour));
69 result += std::string(buffer);
70 }
71 break;
72 case 'H':
73 if (repeat_count <= 2) {
74 char buffer[4] = {0};
75 snprintf(buffer, sizeof(buffer),"%.2d", dt.hour);
76 result += std::string(buffer);
77 }
78 break;
79 case 'h':
80 if (repeat_count == 2) {
81 char buffer[4] = {0};
82 snprintf(buffer, sizeof(buffer),"%.2d", dt.hour);
83 result += std::string(buffer);
84 break;
85 }
86 // %h: Equivalent to %b
87 case 'b':
88 if (repeat_count > 1) break;
89 result += to_str(static_cast<Month>(dt.mon), FormatType::SHORT_NAME);
90 break;
91 case 'B':
92 if (repeat_count > 1) break;
93 result += to_str(static_cast<Month>(dt.mon), FormatType::FULL_NAME);
94 break;
95 case 'c':
96 // Preferred date and time representation for the current locale.
97 // %a %b %e %H:%M:%S %Y
98 if (repeat_count <= 1){
99 char buffer[16];
100 result += to_str(day_of_week(dt.year, dt.mon, dt.day), FormatType::SHORT_NAME);
101 result += " ";
102 result += to_str(static_cast<Month>(dt.mon), FormatType::SHORT_NAME);
103 result += " ";
104 // added %e
105 std::fill(buffer, buffer + sizeof(buffer), '\0');
106 snprintf(buffer, sizeof(buffer),"%2d ", dt.day);
107 result += std::string(buffer);
108 // added %H:%M:%S
109 std::fill(buffer, buffer + sizeof(buffer), '\0');
110 snprintf(buffer, sizeof(buffer), "%.2d:%.2d:%.2d ", dt.hour, dt.min, dt.sec);
111 result += std::string(buffer);
112 // added %Y
113 result += std::to_string(dt.year);
114 }
115 break;
116 case 'C':
117 if (repeat_count > 1) break;
118 result += std::to_string(dt.year/100);
119 break;
120 case 'd':
121 if (repeat_count < 2) {
122 char buffer[4] = {0};
123 snprintf(buffer, sizeof(buffer),"%.2d", dt.day);
124 result += std::string(buffer);
125 }
126 break;
127 case 'D':
128 if (repeat_count == 1) {
129 // %m/%d/%y
130 char buffer[16] = {0};
131 snprintf(buffer, sizeof(buffer), "%.2d/%.2d/%.2d", dt.mon, dt.day, (int)(dt.year % 100LL));
132 result += std::string(buffer);
133 } else
134 if (repeat_count == 2) {
135 char buffer[4] = {0};
136 snprintf(buffer, sizeof(buffer),"%.2d", dt.day);
137 result += std::string(buffer);
138 }
139 break;
140 case 'e':
141 if (repeat_count == 1) {
142 char buffer[4] = {0};
143 snprintf(buffer, sizeof(buffer),"%2d", dt.day);
144 result += std::string(buffer);
145 }
146 break;
147 case 'E':
148 // %E: Modifier for alternative ("era-based") format.

150 break;
151 case 'F':
152 if (repeat_count == 1) {
153 // %Y-%m-%d ISO 8601 date format

155 if (dt.year <= 9999 || dt.year >= 0) {

224 break;

234 if (repeat_count == 1) {
235 char buffer[16] = {0};
236 if (dt.hour < 12) snprintf(buffer, sizeof(buffer), "%.2d:%.2d:%.2d AM", hour24_to_12(dt.hour), dt.min, dt.sec);

240 }
241 break;

243 // %H:%M

245 char buffer[8] = {0};
246 snprintf(buffer, sizeof(buffer), "%.2d:%.2d", dt.hour, dt.min);
247 result += std::string(buffer);
248 }
249 break;

251 if (repeat_count == 1) {
252 result += std::to_string(ts);

256 result += std::to_string(dt.ms);
257 break;

262 char buffer[4] = {0};
263 snprintf(buffer, sizeof(buffer), "%.2d", dt.sec);
264 result += std::string(buffer);

266 if (repeat_count == 3) {
267 result += std::to_string(dt.ms);
268 break;
269 }
270 break;

272 if (repeat_count > 1) break;
273 result += "\t";
274 break;
275 case 'T':
276 // %H:%M:%S
277 if (repeat_count == 1) {
278 char buffer[16] = {0};
279 snprintf(buffer, sizeof(buffer), "%.2d:%.2d:%.2d", dt.hour, dt.min, dt.sec);
280 result += std::string(buffer);
281 }
282 break;
283 case 'u':
284 if (repeat_count == 1) {
285 // Day of the week as a decimal number (1 to 7, Monday being 1).
286 int dw = day_of_week(dt.year, dt.mon, dt.day);
287 if (dw == 0) dw = 7;
288 result += std::to_string(dw);
289 }
290 break;
291 case 'U':
292 // Week number of the current year (00 to 53, starting with the first Sunday as week 01).
293 break;
294 case 'V':
295 // ISO 8601 week number of the current year (01 to 53, with specific rules).
296 break;
297 case 'w':
298 // Day of the week as a decimal number (0 to 6, Sunday being 0).
299 if (repeat_count == 1) {
300 result += std::to_string(day_of_week(dt.year, dt.mon, dt.day));
301 } else
302 if (repeat_count == 3) {
303 result += to_str(day_of_week(dt.year, dt.mon, dt.day), FormatType::SHORT_NAME);
304 }
305 break;
306 case 'W':
307 // Week number of the current year (00 to 53, starting with the first Monday as week 01).
308 if (repeat_count == 3) {
309 result += to_str(day_of_week(dt.year, dt.mon, dt.day), FormatType::UPPERCASE_NAME);
310 }
311 break;
312 case 'x':
313 // Preferred date representation for the current locale without the time.
314 break;
315 case 'X':
316 // Preferred time representation for the current locale without the date.
317 break;
318 case 'y':
319 if (repeat_count == 1) {
320 result += std::to_string(dt.year % 100);
321 }
322 break;
323 case 'Y':
324 if (repeat_count == 1) {
325 result += std::to_string(dt.year);
326 } else
327 if (repeat_count == 6) {
328 char buffer[32] = {0};
329 const int64_t mega_years = dt.year / 1000000;
330 const int64_t millennia = (dt.year - mega_years * 1000000) / 1000;
331 const int64_t centuries = dt.year - mega_years * 1000000 - millennia * 1000;
332 if (mega_years) {
333 if (millennia) {
334 snprintf(buffer, sizeof(buffer), "%lldM%lldK%.3lld", mega_years, std::abs(millennia), std::abs(centuries));
335 } else {
336 snprintf(buffer, sizeof(buffer), "%lldM%.3lld", mega_years, std::abs(centuries));
337 }
338 } else
339 if (millennia) {
340 snprintf(buffer, sizeof(buffer), "%lldK%.3lld", millennia, std::abs(centuries));
341 } else {
342 snprintf(buffer, sizeof(buffer), "%.4lld", dt.year);
343 }
344 result += std::string(buffer);
345 } else
346 if (repeat_count == 4) {
347 char buffer[8] = {0};
348 snprintf(buffer, sizeof(buffer), "%.4d", (int)(dt.year % 10000));
349 result += std::string(buffer);
350 } else
351 if (repeat_count == 2) {
352 char buffer[8] = {0};
353 snprintf(buffer, sizeof(buffer), "%.2d", (int)(dt.year % 100));
354 result += std::string(buffer);
355 }
356 break;
357 case 'z':
358 // +hhmm or -hhmm numeric timezone offset from UTC.
359 if (repeat_count == 1) {
360 TimeZoneStruct tz = to_time_zone_struct(utc_offset);
361 char buffer[16] = {0};
362 if (tz.is_positive) snprintf(buffer, sizeof(buffer), "+%.2d%.2d", tz.hour, tz.min);
363 else snprintf(buffer, sizeof(buffer), "-%.2d%.2d", tz.hour, tz.min);
364 result += std::string(buffer);
365 }
366 break;
367 case 'Z':
368 // Timezone name or abbreviation.
369
370 result += "UTC";
371 break;
372 case '+':
373 // Date and time in date(1) format (not supported in glibc2).
374 // Tue Jun 4 04:07:43 UTC 2024
375 break;
376 };

405 template<class T = ts_t>
406 const std::string to_string(
407 const std::string& format_str,
408 T timestamp,
409 tz_t utc_offset = 0) {
410 std::string result;
411 if (format_str.empty()) return result;
412 DateTimeStruct dt = to_date_time<DateTimeStruct>(timestamp);
413
414 bool is_command = false;
415 size_t repeat_count = 0;
416 char last_char = format_str[0];
417 if (last_char != '%') result += last_char;
418 for (size_t i = 0; i < format_str.size(); ++i) {
419 const char& current_char = format_str[i];
420 if (!is_command) {
421 if (current_char == '%') {
422 ++repeat_count;
423 if (repeat_count == 2) {
424 result += current_char;
425 repeat_count = 0;
426 }
427 continue;
428 }
429 if (!repeat_count) {
430 result += current_char;
431 continue;
432 }
433 last_char = current_char;
434 is_command = true;
435 continue;
436 }
437 if (last_char == current_char) {
438 ++repeat_count;
439 continue;
440 }
441 process_format_impl(last_char, repeat_count, timestamp, utc_offset, dt, result);
442 repeat_count = 0;
443 is_command = false;
444 --i;
445 }
446 if (is_command) {
447 process_format_impl(last_char, repeat_count, timestamp, utc_offset, dt, result);
448 }
449 return result;
450 }

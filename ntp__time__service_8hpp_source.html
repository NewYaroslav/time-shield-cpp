108 void run_loop() {
109 const auto sleep_interval = m_interval;
110 bool is_first = m_measure_immediately;
111 while (!m_is_stop_requested.load()) {
112 if (is_first) {
113 do_measure();
114 is_first = false;
115 } else {
116 std::unique_lock<std::mutex> lk(m_cv_mtx);
117 m_cv.wait_for(lk, sleep_interval, [this]() {
118 return m_is_stop_requested.load() || m_is_force_requested.load();
119 });
120 if (m_is_stop_requested.load()) {
121 break;
122 }
123 m_is_force_requested.store(false);
124 do_measure();
125 }
126 }
127 m_is_running.store(false);

134 const int64_t now = now_realtime_us();

136 m_last_success_realtime_us.store(now);

191 bool init(std::chrono::milliseconds interval, bool measure_immediately = true) {
192 std::unique_ptr<RunnerT> local_runner;
193 {
194 std::lock_guard<std::mutex> lk(m_mtx);
195 if (is_running_locked()) {
196 return true;
197 }
198 if (interval.count() <= 0) {
199 interval = std::chrono::milliseconds(1);
200 }
201 m_interval = interval;
202 m_measure_immediately = measure_immediately;
203
204 local_runner = build_runner_locked();
205 if (!local_runner) {
206 return false;
207 }
208 }
209
210 if (!local_runner->start(m_interval, m_measure_immediately)) {
211 return false;
212 }
213
214 bool is_ok = false;
215 try {
216 is_ok = local_runner->measure_now();
217 } catch (...) {
218 is_ok = false;
219 }
220
221 {
222 std::lock_guard<std::mutex> lk(m_mtx);
223 m_runner = std::move(local_runner);
224 }
225 return is_ok;
226 }

228 void shutdown() {
229 std::unique_ptr<RunnerT> local_runner;
230 {
231 std::lock_guard<std::mutex> lk(m_mtx);
232 if (!m_runner) {
233 return;
234 }
235 local_runner = std::move(m_runner);
236 }
237 try {
238 local_runner->stop();
239 } catch (...) {
240 // no-throw
241 }
242 }

376 bool apply_config_now() {
377 std::unique_ptr<RunnerT> new_runner;
378 std::unique_ptr<RunnerT> old_runner;
379 std::chrono::milliseconds interval;
380 bool measure_immediately = true;
381 {
382 std::lock_guard<std::mutex> lk(m_mtx);
383 new_runner = build_runner_locked();
384 if (!new_runner) {
385 return false;
386 }
387 interval = m_interval;
388 measure_immediately = m_measure_immediately;
389 old_runner = std::move(m_runner);
390 }
391
392 if (old_runner) {
393 try {
394 old_runner->stop();
395 } catch (...) {
396 }
397 }
398
399 if (!new_runner->start(interval, measure_immediately)) {
400 return false;
401 }
402
403 bool is_ok = false;
404 try {
405 is_ok = new_runner->measure_now();
406 } catch (...) {
407 is_ok = false;
408 }
409
410 {
411 std::lock_guard<std::mutex> lk(m_mtx);
412 m_runner = std::move(new_runner);
413 }
414 return is_ok;
415 }

417 private:
418 bool is_running_locked() const noexcept {
419 return m_runner && m_runner->running();
420 }
421
422 std::unique_ptr<RunnerT> build_runner_locked() {
423 std::vector<NtpServerConfig> servers;
424 if (m_has_custom_servers) {
425 servers = m_servers;
426 } else {
427 servers = NtpClientPool::build_default_servers();
428 }
429
430 NtpPoolConfig cfg = m_has_custom_pool_cfg ? m_pool_cfg : NtpPoolConfig{};
431 NtpClientPool pool(cfg);
432 pool.set_servers(std::move(servers));
433
434 std::unique_ptr<RunnerT> runner;
435 try {
436 runner.reset(new RunnerT(std::move(pool)));
437 } catch (...) {
438 return nullptr;
439 }
440 return runner;
441 }
442
443 private:
444 mutable std::mutex m_mtx;
445 std::chrono::milliseconds m_interval{std::chrono::seconds(30)};
446 bool m_measure_immediately{true};
447
448 bool m_has_custom_servers{false};
449 std::vector<NtpServerConfig> m_servers;
450
451 bool m_has_custom_pool_cfg{false};
452 NtpPoolConfig m_pool_cfg{};
453
454 std::unique_ptr<RunnerT> m_runner;
455 };

664 static bool try_parse_iso8601_buffer(const char* data, std::size_t size, DateTime& out) noexcept {
665 if (data == nullptr) {
666 return false;
667 }
668 DateTimeStruct dt = create_date_time_struct(0);
669 TimeZoneStruct tz = create_time_zone_struct(0, 0, true);
670 if (!time_shield::parse_iso8601(data, size, dt, tz)) {
671 return false;
672 }
673 const tz_t offset = time_zone_struct_to_offset(tz);
674 if (!is_valid_tz_offset(offset)) {
675 return false;
676 }
677 const ts_ms_t utc_ms = dt_to_timestamp_ms(dt) - offset_to_ms(offset);
678 out = from_unix_ms(utc_ms, offset);
679 return true;
680 }
681
682 static DateTime parse_iso8601_buffer(const char* data, std::size_t size) {
683 DateTime result;
684 if (!try_parse_iso8601_buffer(data, size, result)) {
685 throw std::invalid_argument("Invalid ISO8601 datetime");
686 }
687 return result;
688 }

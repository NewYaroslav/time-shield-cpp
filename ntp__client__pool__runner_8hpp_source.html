114 void run_loop(std::chrono::milliseconds interval, bool measure_immediately) {
115 bool is_first = measure_immediately;
116 while (!m_is_stop_requested.load()) {
117 if (is_first) {
118 do_measure();
119 is_first = false;
120 } else {
121 std::unique_lock<std::mutex> lk(m_cv_mtx);
122 m_cv.wait_for(lk, interval, [this]() {
123 return m_is_stop_requested.load() || m_is_force_requested.load();
124 });
125 if (m_is_stop_requested.load()) {
126 break;
127 }
128 m_is_force_requested.store(false);
129 do_measure();
130 }
131 }
132 m_is_running.store(false);
133 }
134
135 bool do_measure() {
136 bool is_ok = false;
137 try {
138 std::lock_guard<std::mutex> lk(m_pool_mtx);
139 is_ok = m_pool.measure();
140 } catch (...) {
141 is_ok = false;
142 }
143
144 m_measure_count.fetch_add(1);
145 if (!is_ok) {
146 m_fail_count.fetch_add(1);
147 }
148
149 m_last_measure_ok.store(is_ok);
150
151 const int64_t now = now_realtime_us();
152 m_last_update_realtime_us.store(now);
153 if (is_ok) {
154 m_last_success_realtime_us.store(now);
155 }
156
157 return is_ok;
158 }

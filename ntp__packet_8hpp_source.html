61 static inline uint64_t ntp_frac_to_us(uint32_t frac_net) noexcept {
62 const uint64_t frac = static_cast<uint64_t>(ntohl(frac_net));
63 return (frac * 1000000ULL) >> 32;

66 static inline bool ntp_ts_to_unix_us(uint32_t sec_net, uint32_t frac_net, uint64_t& out_us) noexcept {
67 static const int64_t NTP_TIMESTAMP_DELTA = 2208988800ll;
68 const int64_t sec = static_cast<int64_t>(ntohl(sec_net)) - NTP_TIMESTAMP_DELTA;
69 if (sec < 0) return false;
70 out_us = static_cast<uint64_t>(sec) * 1000000ULL + ntp_frac_to_us(frac_net);
71 return true;
72 }

74 static inline void fill_client_packet(NtpPacket& pkt, uint64_t now_us) {
75 std::memset(&pkt, 0, sizeof(pkt));
76 pkt.li_vn_mode = static_cast<uint8_t>((0 << 6) | (3 << 3) | 3); // LI=0, VN=3, Mode=3
77
78 const uint64_t sec = now_us / 1000000 + 2208988800ULL;
79 const uint64_t frac = ((now_us % 1000000) * 0x100000000ULL) / 1000000;
80
81 pkt.tx_ts_sec = htonl(static_cast<uint32_t>(sec));
82 pkt.tx_ts_frac = htonl(static_cast<uint32_t>(frac));
83 }

85 static inline bool parse_server_packet(const NtpPacket& pkt,
86 uint64_t arrival_us,
87 int64_t& offset_us,
88 int64_t& delay_us,
89 int& stratum,
90 int& out_error_code) noexcept {
91 const uint8_t li = ntp_li(pkt.li_vn_mode);
92 const uint8_t vn = ntp_vn(pkt.li_vn_mode);
93 const uint8_t mode = ntp_mode(pkt.li_vn_mode);
94
95 if (mode != 4) {
96 out_error_code = NTP_E_BAD_MODE;
97 return false;
98 }
99 if (vn < 3 || vn > 4) {
100 out_error_code = NTP_E_BAD_VERSION;
101 return false;
102 }
103 if (li == 3) {
104 out_error_code = NTP_E_BAD_LI;
105 return false;
106 }
107 if (pkt.stratum == 0) {
108 out_error_code = NTP_E_KOD;
109 return false;
110 }
111 if (pkt.stratum >= 16) {
112 out_error_code = NTP_E_BAD_STRATUM;
113 return false;
114 }
115
116 uint64_t originate_us = 0;
117 uint64_t receive_us = 0;
118 uint64_t transmit_us = 0;
119
120 if (!ntp_ts_to_unix_us(pkt.orig_ts_sec, pkt.orig_ts_frac, originate_us)) {
121 out_error_code = NTP_E_BAD_TS;
122 return false;
123 }
124 if (!ntp_ts_to_unix_us(pkt.recv_ts_sec, pkt.recv_ts_frac, receive_us)) {
125 out_error_code = NTP_E_BAD_TS;
126 return false;
127 }
128 if (!ntp_ts_to_unix_us(pkt.tx_ts_sec, pkt.tx_ts_frac, transmit_us)) {
129 out_error_code = NTP_E_BAD_TS;
130 return false;
131 }
132
133 const int64_t t1 = static_cast<int64_t>(originate_us);
134 const int64_t t2 = static_cast<int64_t>(receive_us);
135 const int64_t t3 = static_cast<int64_t>(transmit_us);
136 const int64_t t4 = static_cast<int64_t>(arrival_us);
137
138 if (t3 < t2) {
139 out_error_code = NTP_E_BAD_TS;
140 return false;
141 }
142
143 offset_us = ((t2 - t1) + (t3 - t4)) / 2;
144 delay_us = (t4 - t1) - (t3 - t2);
145 if (delay_us < 0) {
146 out_error_code = NTP_E_BAD_TS;
147 return false;
148 }
149 stratum = pkt.stratum;
150 return true;
151 }

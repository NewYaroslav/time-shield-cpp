74 inline std::string trim_copy_ascii(const std::string& s) {
75 size_t b = 0;
76 size_t e = s.size();
77 while (b < e && std::isspace(static_cast<unsigned char>(s[b])) != 0) ++b;
78 while (e > b && std::isspace(static_cast<unsigned char>(s[e - 1])) != 0) --e;
79 return s.substr(b, e - b);
80 }

213 TIME_SHIELD_CONSTEXPR inline bool is_ascii_space(char c) noexcept {

219 return c >= '0' && c <= '9';
220 }

224 while (p < end && is_ascii_space(*p)) {

231 TIME_SHIELD_CONSTEXPR inline bool parse_2digits(const char*& p, const char* end, int& out) noexcept {
232 if (end - p < 2) {
233 return false;
234 }
235 const char a = p[0];
236 const char b = p[1];
237 if (!is_ascii_digit(a) || !is_ascii_digit(b)) {
238 return false;
239 }
240 out = (a - '0') * 10 + (b - '0');
241 p += 2;
242 return true;
243 }

247 TIME_SHIELD_CONSTEXPR inline bool parse_4digits_year(const char*& p, const char* end, year_t& out) noexcept {
248 if (end - p < 4) {
249 return false;
250 }
251 const char a = p[0], b = p[1], c = p[2], d = p[3];
252 if (!is_ascii_digit(a) || !is_ascii_digit(b) || !is_ascii_digit(c) || !is_ascii_digit(d)) {
253 return false;
254 }
255 const int v = (a - '0') * 1000 + (b - '0') * 100 + (c - '0') * 10 + (d - '0');
256 out = static_cast<year_t>(v);
257 p += 4;
258 return true;
259 }

264 TIME_SHIELD_CONSTEXPR inline bool parse_fraction_to_ms(const char*& p, const char* end, int& ms_out) noexcept {
265 if (p >= end || !is_ascii_digit(*p)) {
266 return false;
267 }
268
269 // Read up to 9 digits, but we only need first 3 for ms.
270 int ms = 0;
271 int digits = 0;
272
273 while (p < end && is_ascii_digit(*p) && digits < 9) {
274 if (digits < 3) {
275 ms = ms * 10 + (*p - '0');
276 }
277 ++digits;
278 ++p;
279 }
280
281 // If there are more than 9 digits, that is not supported (keep strict).
282 if (p < end && is_ascii_digit(*p)) {
283 return false;
284 }
285
286 // Scale if fewer than 3 digits.
287 if (digits == 1) ms *= 100;
288 else if (digits == 2) ms *= 10;
289
290 ms_out = ms;
291 return true;
292 }

564 inline bool parse_time_zone(const char* data, std::size_t length, TimeZoneStruct& tz) noexcept {
565 if (!data) {
566 return false;
567 }
568
569 if (length == 0) {
570 tz.hour = 0;
571 tz.min = 0;
572 tz.is_positive = true;
573 return true;
574 }
575
576 if (length == 1 && (data[0] == 'Z' || data[0] == 'z')) {
577 tz.hour = 0;
578 tz.min = 0;
579 tz.is_positive = true;
580 return true;
581 }
582
583 if (length != 6) {
584 return false;
585 }
586
587 const char sign = data[0];
588 if (sign != '+' && sign != '-') {
589 return false;
590 }
591 if (data[3] != ':') {
592 return false;
593 }
594 if (!detail::is_ascii_digit(data[1]) || !detail::is_ascii_digit(data[2]) ||
595 !detail::is_ascii_digit(data[4]) || !detail::is_ascii_digit(data[5])) {
596 return false;
597 }
598
599 tz.is_positive = (sign == '+');
600 tz.hour = (data[1] - '0') * 10 + (data[2] - '0');
601 tz.min = (data[4] - '0') * 10 + (data[5] - '0');
602
603 return is_valid_time_zone(tz);
604 }

622//------------------------------------------------------------------------------
623// ISO8601 parsing (C-style, no regex, no allocations)
624//------------------------------------------------------------------------------
625
642 inline bool parse_iso8601(const char* input, std::size_t length,
643 DateTimeStruct& dt, TimeZoneStruct& tz) noexcept {
644 if (!input) {
645 return false;
646 }
647
648 const char* p = input;
649 const char* end = input + length;
650
651 detail::skip_spaces(p, end);
652
653 dt = create_date_time_struct(0);
654 tz = create_time_zone_struct(0, 0);
655 tz.is_positive = true;
656
657 // ---- Date: YYYY<sep>MM<sep>DD
658 if (!detail::parse_4digits_year(p, end, dt.year)) {
659 return false;
660 }
661 if (p >= end) {
662 return false;
663 }
664 const char sep1 = *p;
665 if (sep1 != '-' && sep1 != '/' && sep1 != '.') {
666 return false;
667 }
668 ++p;
669
670 if (!detail::parse_2digits(p, end, dt.mon)) {
671 return false;
672 }
673 if (p >= end) {
674 return false;
675 }
676 const char sep2 = *p;
677 if (sep2 != '-' && sep2 != '/' && sep2 != '.') {
678 return false;
679 }
680 ++p;
681
682 if (!detail::parse_2digits(p, end, dt.day)) {
683 return false;
684 }
685
686 // Date-only?
687 {
688 const char* q = p;
689 detail::skip_spaces(q, end);
690 if (q == end) {
691 // dt already has time=0 ms=0
692 return is_valid_date_time(dt);
693 }
694 }
695
696 // ---- Date/time separator: 'T' or whitespace
697 if (p >= end) {
698 return false;
699 }
700
701 if (*p == 'T' || *p == 't') {
702 ++p;
703 } else
704 if (detail::is_ascii_space(*p)) {
705 // allow one or more spaces
706 detail::skip_spaces(p, end);
707 } else {
708 return false;
709 }
710
711 // ---- Time: hh:mm[:ss][.frac]
712 if (!detail::parse_2digits(p, end, dt.hour)) {
713 return false;
714 }
715 if (p >= end || *p != ':') {
716 return false;
717 }
718 ++p;
719
720 if (!detail::parse_2digits(p, end, dt.min)) {
721 return false;
722 }
723
724 dt.sec = 0;
725 dt.ms = 0;
726
727 // Optional :ss
728 if (p < end && *p == ':') {
729 ++p;
730 if (!detail::parse_2digits(p, end, dt.sec)) {
731 return false;
732 }
733 }
734
735 // Optional .fraction (allowed only if we had seconds in original regex,
736 // but we accept it when seconds are present; for hh:mm (no seconds) we keep it strict).
737 if (p < end && *p == '.') {
738 // require seconds field to exist (avoid accepting YYYY-MM-DDThh:mm.xxx)
739 if (dt.sec == 0) {
740 // Ambiguous: could be "hh:mm.fff" which is not in your original formats.
741 // Keep strict to preserve behavior.
742 return false;
743 }
744
745 ++p;
746 int ms = 0;
747 if (!detail::parse_fraction_to_ms(p, end, ms)) {
748 return false;
749 }
750 dt.ms = ms;
751 }
752
753 // ---- Optional timezone: [spaces] (Z | HH:MM)
754 detail::skip_spaces(p, end);
755
756 if (p < end) {
757 if (*p == 'Z' || *p == 'z') {
758 tz.hour = 0;
759 tz.min = 0;
760 tz.is_positive = true;
761 ++p;
762 } else if (*p == '+' || *p == '-') {
763 // need 6 chars
764 if (static_cast<std::size_t>(end - p) < 6) {
765 return false;
766 }
767 if (!parse_time_zone(p, 6, tz)) {
768 return false;
769 }
770 p += 6;
771 }
772 }
773
774 detail::skip_spaces(p, end);
775 if (p != end) {
776 return false;
777 }
778
779 return is_valid_date_time(dt);
780 }

1105 template<class T = int>
1106 inline bool sec_of_day(const std::string& str, T& sec) {
1107 if (str.empty()) return false;
1108
1109 const char* p = str.c_str();
1110 int parts[3] = {0, 0, 0}; // hour, minute, second
1111 int idx = 0;
1112
1113 while (*p && idx < 3) {
1114 // Parse integer
1115 int value = 0;
1116 bool has_digit = false;
1117
1118 while (*p >= '0' && *p <= '9') {
1119 has_digit = true;
1120 value = value * 10 + (*p - '0');
1121 ++p;
1122 }
1123
1124 if (!has_digit) return false;
1125 parts[idx++] = value;
1126
1127 // Expect colon or end
1128 if (*p == ':') {
1129 ++p;
1130 } else if (*p == '\0') {
1131 break;
1132 } else {
1133 return false; // unexpected character
1134 }
1135 }
1136
1137 if (idx == 0) return false;
1138 if (!is_valid_time(parts[0], parts[1], parts[2])) return false;
1139
1140 sec = static_cast<T>(sec_of_day(parts[0], parts[1], parts[2]));
1141 return true;
1142 }

1158 return value;

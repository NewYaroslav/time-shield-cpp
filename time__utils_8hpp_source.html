148# if defined(_WIN32)

150 if (GetProcessTimes(GetCurrentProcess(), &create_time, &exit_time, &kernel_time, &user_time)) {
151 ULARGE_INTEGER li{};
152 li.LowPart = user_time.dwLowDateTime;
153 li.HighPart = user_time.dwHighDateTime;
154 return static_cast<double>(li.QuadPart) / 10000000.0;
155 }
156# elif defined(__unix__) || defined(__unix) || defined(unix) || (defined(__APPLE__) && defined(__MACH__))
157 // AIX, BSD, Cygwin, HP-UX, Linux, OSX, and Solaris
158# if defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0)
159 clockid_t id = (clockid_t)-1;
160# if defined(_POSIX_CPUTIME) && (_POSIX_CPUTIME > 0)
161 if (clock_getcpuclockid(0, &id) != 0) {
162# if defined(CLOCK_PROCESS_CPUTIME_ID)
163 id = CLOCK_PROCESS_CPUTIME_ID;
164# elif defined(CLOCK_VIRTUAL)
165 id = CLOCK_VIRTUAL;
166# endif
167 }
168# elif defined(CLOCK_PROCESS_CPUTIME_ID)
169 id = CLOCK_PROCESS_CPUTIME_ID;
170# elif defined(CLOCK_VIRTUAL)
171 id = CLOCK_VIRTUAL;
172# endif
173 if (id != (clockid_t)-1) {
174 struct timespec ts;
175 if (clock_gettime(id, &ts) == 0) {
176 return static_cast<double>(ts.tv_sec) + static_cast<double>(ts.tv_nsec) / 1e9;
177 }
178 }
179# endif
180
181# if defined(RUSAGE_SELF)
182 struct rusage usage{};
183 if (getrusage(RUSAGE_SELF, &usage) == 0) {
184 return static_cast<double>(usage.ru_utime.tv_sec) + static_cast<double>(usage.ru_utime.tv_usec) / 1e6;
185 }
186# endif
187
188# if defined(_SC_CLK_TCK)
189 struct tms t{};
190 if (times(&t) != (clock_t)-1) {
191 return static_cast<double>(t.tms_utime) / static_cast<double>(sysconf(_SC_CLK_TCK));
192 }
193# endif

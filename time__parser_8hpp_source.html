174  } else {
175  return false;

178  // Парсинг времени и часового пояса
179  if (std::regex_search(input, time_match, time_regex)) {
180  dt.hour = std::stoi(time_match[1].str());
181  dt.min = std::stoi(time_match[2].str());
182  dt.sec = std::stoi(time_match[3].str());

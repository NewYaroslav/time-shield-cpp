668 throw std::invalid_argument("Invalid date-time combination");
669 }

671 int64_t secs = 0;
672 uint64_t years = (static_cast<int64_t>(MAX_YEAR) - year);
673
674 const int64_t n_400_years = years / 400;
675 secs += n_400_years * SEC_PER_400_YEARS;
676 years -= n_400_years * 400;
677

679 secs += n_100_years * SEC_PER_100_YEARS;
680 years -= n_100_years * 100;
681
682 const int64_t n_4_years = years / 4;
683 secs += n_4_years * SEC_PER_4_YEARS;
684 years -= n_4_years * 4;

688 // 9223372029693630000 - значение на момент 292277024400 от 2000 года
689 // Такое значение приводит к неправильному вычислению умножения n_400_years * SEC_PER_400_YEARS

692 constexpr int64_t BIAS_2000 = 946684800LL;

695 secs += BIAS_2000;

698 hour == 0 && min == 0 &&
699 sec == 0) {

1700 TIME_SHIELD_CONSTEXPR inline const ts_t start_of_year(ts_t ts) noexcept {
1701 constexpr ts_t BIAS_2100 = 4102444800;
1702 if (ts < BIAS_2100) {
1703 constexpr ts_t SEC_PER_YEAR_X2 = SEC_PER_YEAR * 2;
1704 ts_t year_start_ts = ts % SEC_PER_4_YEARS;
1705 if (year_start_ts < SEC_PER_YEAR) {
1706 return ts - year_start_ts;
1707 } else
1708 if (year_start_ts < SEC_PER_YEAR_X2) {
1709 return ts + SEC_PER_YEAR - year_start_ts;
1710 } else
1711 if (year_start_ts < (SEC_PER_YEAR_X2 + SEC_PER_LEAP_YEAR)) {
1712 return ts + SEC_PER_YEAR_X2 - year_start_ts;
1713 }
1714 return ts + (SEC_PER_YEAR_X2 + SEC_PER_LEAP_YEAR) - year_start_ts;
1715 }
1716
1717 constexpr ts_t BIAS_2000 = 946684800;
1718 ts_t secs = ts - BIAS_2000;
1719
1720 ts_t offset_y400 = secs % SEC_PER_400_YEARS;
1721 ts_t start_ts = secs - offset_y400 + BIAS_2000;
1722 secs = offset_y400;
1723
1724 if (secs >= SEC_PER_FIRST_100_YEARS) {
1725 secs -= SEC_PER_FIRST_100_YEARS;
1726 start_ts += SEC_PER_FIRST_100_YEARS;
1727 while (secs >= SEC_PER_100_YEARS) {
1728 secs -= SEC_PER_100_YEARS;
1729 start_ts += SEC_PER_100_YEARS;
1730 }
1731
1732 constexpr ts_t SEC_PER_4_YEARS_V2 = 4 * SEC_PER_YEAR;
1733 if (secs >= SEC_PER_4_YEARS_V2) {
1734 secs -= SEC_PER_4_YEARS_V2;
1735 start_ts += SEC_PER_4_YEARS_V2;
1736 } else {
1737 start_ts += secs - secs % SEC_PER_YEAR;
1738 return start_ts;
1739 }
1740 }
1741
1742 ts_t offset_4y = secs % SEC_PER_4_YEARS;
1743 start_ts += secs - offset_4y;
1744 secs = offset_4y;
1745
1746 if (secs >= SEC_PER_LEAP_YEAR) {
1747 secs -= SEC_PER_LEAP_YEAR;
1748 start_ts += SEC_PER_LEAP_YEAR;
1749 start_ts += secs - secs % SEC_PER_YEAR;
1750 }
1751 return start_ts;
1752 }

1869 TIME_SHIELD_CONSTEXPR inline ts_t end_of_year(ts_t ts = ts()) {
1870 constexpr ts_t BIAS_2100 = 4102444800;
1871 if (ts < BIAS_2100) {
1872 constexpr ts_t SEC_PER_YEAR_X2 = SEC_PER_YEAR * 2;
1873 constexpr ts_t SEC_PER_YEAR_X3 = SEC_PER_YEAR * 3;
1874 constexpr ts_t SEC_PER_YEAR_X3_V2 = SEC_PER_YEAR_X2 + SEC_PER_LEAP_YEAR;
1875 ts_t year_end_ts = ts % SEC_PER_4_YEARS;
1876 if (year_end_ts < SEC_PER_YEAR) {
1877 return ts + SEC_PER_YEAR - year_end_ts - 1;
1878 } else
1879 if (year_end_ts < SEC_PER_YEAR_X2) {
1880 return ts + SEC_PER_YEAR_X2 - year_end_ts - 1;
1881 } else
1882 if (year_end_ts < SEC_PER_YEAR_X3_V2) {
1883 return ts + SEC_PER_YEAR_X3_V2 - year_end_ts - 1;
1884 }
1885 return ts + (SEC_PER_YEAR_X3 + SEC_PER_LEAP_YEAR) - year_end_ts - 1;
1886 }
1887
1888 constexpr ts_t BIAS_2000 = 946684800;
1889 ts_t secs = ts - BIAS_2000;
1890
1891 ts_t offset_y400 = secs % SEC_PER_400_YEARS;
1892 ts_t end_ts = secs - offset_y400 + BIAS_2000;
1893 secs = offset_y400;
1894
1895 if (secs >= SEC_PER_FIRST_100_YEARS) {
1896 secs -= SEC_PER_FIRST_100_YEARS;
1897 end_ts += SEC_PER_FIRST_100_YEARS;
1898 while (secs >= SEC_PER_100_YEARS) {
1899 secs -= SEC_PER_100_YEARS;
1900 end_ts += SEC_PER_100_YEARS;
1901 }
1902
1903 constexpr ts_t SEC_PER_4_YEARS_V2 = 4 * SEC_PER_YEAR;
1904 if (secs >= SEC_PER_4_YEARS_V2) {
1905 secs -= SEC_PER_4_YEARS_V2;
1906 end_ts += SEC_PER_4_YEARS_V2;
1907 } else {
1908 end_ts += secs - secs % SEC_PER_YEAR;
1909 return end_ts + SEC_PER_YEAR - 1;
1910 }
1911 }
1912
1913 ts_t offset_4y = secs % SEC_PER_4_YEARS;
1914 end_ts += secs - offset_4y;
1915 secs = offset_4y;
1916
1917 if (secs >= SEC_PER_LEAP_YEAR) {
1918 secs -= SEC_PER_LEAP_YEAR;
1919 end_ts += SEC_PER_LEAP_YEAR;
1920 end_ts += secs - secs % SEC_PER_YEAR;
1921 end_ts += SEC_PER_YEAR;
1922 } else {
1923 end_ts += SEC_PER_LEAP_YEAR;
1924 }
1925 return end_ts - 1;
1926 }

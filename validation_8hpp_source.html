49 TIME_SHIELD_CONSTEXPR const bool is_leap_year_ts(ts_t ts) {
50 // 9223372029693630000 - значение на момент 292277024400 от 2000 года
51 // Такое значение приводит к неправильному вычислению умножения n_400_years * SEC_PER_400_YEARS
52 // Поэтому пришлось снизить до 9223371890843040000
53 constexpr int64_t BIAS_292277022000 = 9223371890843040000LL;
54 constexpr int64_t BIAS_2000 = 946684800LL;
55
56 int64_t y = MAX_YEAR;
57 int64_t secs = -((ts - BIAS_2000) - BIAS_292277022000);
58
59 const int64_t n_400_years = secs / SEC_PER_400_YEARS;
60 secs -= n_400_years * SEC_PER_400_YEARS;
61 y -= n_400_years * 400;
62
63 const int64_t n_100_years = secs / SEC_PER_100_YEARS;
64 secs -= n_100_years * SEC_PER_100_YEARS;
65 y -= n_100_years * 100;
66
67 const int64_t n_4_years = secs / SEC_PER_4_YEARS;
68 secs -= n_4_years * SEC_PER_4_YEARS;
69 y -= n_4_years * 4;
70
71 const int64_t n_1_years = secs / SEC_PER_YEAR;
72 secs -= n_1_years * SEC_PER_YEAR;
73 y -= n_1_years;
74
75 y = secs == 0 ? y : y - 1;
76 return is_leap_year_date(y);
77 }

191 TIME_SHIELD_CONSTEXPR inline const bool is_valid_date(
192 T1 year,
193 T2 month,
194 T2 day) noexcept {
195 if (day > 31 && year <= 31) {
196 return is_valid_date((T1)day, month, (T2)year);
197 }
198 if (year > MAX_YEAR) return false;
199 if (month < 1 || month > 12) return false;
200 if (day < 1 || day > 31) return false;
201 if (month == FEB) {
202 const bool is_leap_year = is_leap_year_date(year);
203 if (is_leap_year && day > 29) return false;
204 if (!is_leap_year && day > 28) return false;
205 } else {
206 switch(month) {
207 case 4:
208 case 6:
209 case 9:
210 case 11:
211 if (day > 30) return false;
212 default:
213 break;
214 };
215 }
216 return true;

46 inline jd_t gregorian_to_jd(double day, int64_t month, int64_t year) noexcept {
47 // Algorithm source (as per your original code): krutov.org Julianday
48 if (month == 1 || month == 2) {
49 year -= 1;
50 month += 12;
51 }
52 const double a = std::floor(static_cast<double>(year) / 100.0);
53 const double b = 2.0 - a + std::floor(a / 4.0);
54 const double jdn = std::floor(365.25 * (static_cast<double>(year) + 4716.0))
55 + std::floor(30.6000001 * (static_cast<double>(month) + 1.0))
56 + day + b - 1524.5;
57 return static_cast<jd_t>(jdn);
58 }

107 inline jdn_t gregorian_to_jdn(uint32_t day, uint32_t month, uint32_t year) noexcept {
108 const uint64_t a = (14ULL - static_cast<uint64_t>(month)) / 12ULL;
109 const uint64_t y = static_cast<uint64_t>(year) + 4800ULL - a;
110 const uint64_t m = static_cast<uint64_t>(month) + 12ULL * a - 3ULL;
111 const uint64_t jdn = static_cast<uint64_t>(day)
112 + (153ULL * m + 2ULL) / 5ULL
113 + 365ULL * y
114 + y / 4ULL
115 - y / 100ULL
116 + y / 400ULL
117 - 32045ULL;
118 return static_cast<jdn_t>(jdn);
119 }

125 inline double moon_phase(fts_t ts) noexcept {
126 double temp = (static_cast<double>(fts_to_jd(ts)) - 2451550.1) / 29.530588853;
127 temp = temp - std::floor(temp);
128 if (temp < 0.0) temp += 1.0;
129 return temp;
130 }

187 long get_unix_year(long ts) {
188 const long BIAS_292277022000 = 9223371890843040000;
189 const long BIAS_2000 = 946684800;
190
191 long y = MAX_YEAR;
192 long secs = -((ts - BIAS_2000) - BIAS_292277022000);
193
194 long n_400_years = secs / SEC_PER_400_YEARS;
195 secs -= n_400_years * SEC_PER_400_YEARS;
196 y -= n_400_years * 400;
197
198 long n_100_years = secs / SEC_PER_100_YEARS;
199 secs -= n_100_years * SEC_PER_100_YEARS;
200 y -= n_100_years * 100;
201
202 long n_4_years = secs / SEC_PER_4_YEARS;
203 secs -= n_4_years * SEC_PER_4_YEARS;
204 y -= n_4_years * 4;
205
206 long n_1_years = secs / SEC_PER_YEAR;
207 secs -= n_1_years * SEC_PER_YEAR;
208 y -= n_1_years;
209
210 y = secs == 0 ? y : y - 1;
211 return y - UNIX_EPOCH;
212 }

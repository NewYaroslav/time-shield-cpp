74 inline std::string trim_copy_ascii(const std::string& s) {
75 size_t b = 0;
76 size_t e = s.size();
77 while (b < e && std::isspace(static_cast<unsigned char>(s[b])) != 0) ++b;
78 while (e > b && std::isspace(static_cast<unsigned char>(s[e - 1])) != 0) --e;
79 return s.substr(b, e - b);
80 }

213 TIME_SHIELD_CONSTEXPR inline bool is_ascii_space(char c) noexcept {
214 return c == ' ' || c == '\t' || c == '\n' || c == '\r' || c == '\f' || c == '\v';
215 }

218 TIME_SHIELD_CONSTEXPR inline bool is_ascii_digit(char c) noexcept {
219 return c >= '0' && c <= '9';
220 }
221
223 TIME_SHIELD_CONSTEXPR inline void skip_spaces(const char*& p, const char* end) noexcept {
224 while (p < end && is_ascii_space(*p)) {
225 ++p;
226 }
227 }

231 TIME_SHIELD_CONSTEXPR inline bool parse_2digits(const char*& p, const char* end, int& out) noexcept {
232 if (end - p < 2) {
233 return false;
234 }
235 const char a = p[0];
236 const char b = p[1];
237 if (!is_ascii_digit(a) || !is_ascii_digit(b)) {
238 return false;
239 }
240 out = (a - '0') * 10 + (b - '0');
241 p += 2;
242 return true;
243 }
244
247 TIME_SHIELD_CONSTEXPR inline bool parse_4digits_year(const char*& p, const char* end, year_t& out) noexcept {
248 if (end - p < 4) {
249 return false;
250 }
251 const char a = p[0], b = p[1], c = p[2], d = p[3];
252 if (!is_ascii_digit(a) || !is_ascii_digit(b) || !is_ascii_digit(c) || !is_ascii_digit(d)) {
253 return false;
254 }
255 const int v = (a - '0') * 1000 + (b - '0') * 100 + (c - '0') * 10 + (d - '0');
256 out = static_cast<year_t>(v);
257 p += 4;
258 return true;
259 }
260
264 TIME_SHIELD_CONSTEXPR inline bool parse_fraction_to_ms(const char*& p, const char* end, int& ms_out) noexcept {
265 if (p >= end || !is_ascii_digit(*p)) {
266 return false;
267 }
268
269 int ms = 0;
270 int digits = 0;
271
272 while (p < end && is_ascii_digit(*p)) {
273 if (digits >= 3) {
274 return false;
275 }
276 ms = ms * 10 + (*p - '0');
277 ++digits;
278 ++p;
279 }
280
281 if (digits == 1) {
282 ms *= 100;
283 } else if (digits == 2) {
284 ms *= 10;
285 }
286
287 ms_out = ms;
288 return true;
289 }

561 inline bool parse_time_zone(const char* data, std::size_t length, TimeZoneStruct& tz) noexcept {
562 if (!data) {
563 return false;
564 }
565
566 if (length == 0) {
567 tz.hour = 0;
568 tz.min = 0;
569 tz.is_positive = true;
570 return true;
571 }
572
573 if (length == 1 && (data[0] == 'Z' || data[0] == 'z')) {
574 tz.hour = 0;
575 tz.min = 0;
576 tz.is_positive = true;
577 return true;
578 }
579
580 if (length != 6) {
581 return false;
582 }
583
584 const char sign = data[0];
585 if (sign != '+' && sign != '-') {
586 return false;
587 }
588 if (data[3] != ':') {
589 return false;
590 }
591 if (!detail::is_ascii_digit(data[1]) || !detail::is_ascii_digit(data[2]) ||
592 !detail::is_ascii_digit(data[4]) || !detail::is_ascii_digit(data[5])) {
593 return false;
594 }
595
596 tz.is_positive = (sign == '+');
597 tz.hour = (data[1] - '0') * 10 + (data[2] - '0');
598 tz.min = (data[4] - '0') * 10 + (data[5] - '0');
599
600 return is_valid_time_zone(tz);
601 }

619//------------------------------------------------------------------------------
620// ISO8601 parsing (C-style, no regex, no allocations)
621//------------------------------------------------------------------------------
622
639 inline bool parse_iso8601(const char* input, std::size_t length,
640 DateTimeStruct& dt, TimeZoneStruct& tz) noexcept {
641 if (!input) {
642 return false;
643 }
644
645 const char* p = input;
646 const char* end = input + length;
647
648 detail::skip_spaces(p, end);
649
650 dt = create_date_time_struct(0);
651 tz = create_time_zone_struct(0, 0);
652 tz.is_positive = true;
653
654 // ---- Date: YYYY<sep>MM<sep>DD
655 if (!detail::parse_4digits_year(p, end, dt.year)) {
656 return false;
657 }
658 if (p >= end) {
659 return false;
660 }
661 const char sep1 = *p;
662 if (sep1 != '-' && sep1 != '/' && sep1 != '.') {
663 return false;
664 }
665 ++p;
666
667 if (!detail::parse_2digits(p, end, dt.mon)) {
668 return false;
669 }
670 if (p >= end) {
671 return false;
672 }
673 const char sep2 = *p;
674 if (sep2 != '-' && sep2 != '/' && sep2 != '.') {
675 return false;
676 }
677 ++p;
678
679 if (!detail::parse_2digits(p, end, dt.day)) {
680 return false;
681 }
682
683 // Date-only?
684 {
685 const char* q = p;
686 detail::skip_spaces(q, end);
687 if (q == end) {
688 // dt already has time=0 ms=0
689 return is_valid_date_time(dt);
690 }
691 }
692
693 // ---- Date/time separator: 'T' or whitespace
694 if (p >= end) {
695 return false;
696 }
697
698 if (*p == 'T' || *p == 't') {
699 ++p;
700 } else
701 if (detail::is_ascii_space(*p)) {
702 // allow one or more spaces
703 detail::skip_spaces(p, end);
704 } else {
705 return false;
706 }
707
708 // ---- Time: hh:mm[:ss][.frac]
709 if (!detail::parse_2digits(p, end, dt.hour)) {
710 return false;
711 }
712 if (p >= end || *p != ':') {
713 return false;
714 }
715 ++p;
716
717 if (!detail::parse_2digits(p, end, dt.min)) {
718 return false;
719 }
720
721 dt.sec = 0;
722 dt.ms = 0;
723 bool has_seconds = false;
724
725 // Optional :ss
726 if (p < end && *p == ':') {
727 ++p;
728 if (!detail::parse_2digits(p, end, dt.sec)) {
729 return false;
730 }
731 has_seconds = true;
732 }
733
734 // Optional .fraction (allowed only if we had seconds in original regex,
735 // but we accept it when seconds are present; for hh:mm (no seconds) we keep it strict).
736 if (p < end && *p == '.') {
737 // require seconds field to exist (avoid accepting YYYY-MM-DDThh:mm.xxx)
738 if (!has_seconds) {
739 // Ambiguous: could be "hh:mm.fff" which is not in your original formats.
740 // Keep strict to preserve behavior.
741 return false;
742 }
743
744 ++p;
745 int ms = 0;
746 if (!detail::parse_fraction_to_ms(p, end, ms)) {
747 return false;
748 }
749 dt.ms = ms;
750 }
751
752 // ---- Optional timezone: [spaces] (Z | HH:MM)
753 detail::skip_spaces(p, end);
754
755 if (p < end) {
756 if (*p == 'Z' || *p == 'z') {
757 tz.hour = 0;
758 tz.min = 0;
759 tz.is_positive = true;
760 ++p;
761 } else if (*p == '+' || *p == '-') {
762 // need 6 chars
763 if (static_cast<std::size_t>(end - p) < 6) {
764 return false;
765 }
766 if (!parse_time_zone(p, 6, tz)) {
767 return false;
768 }
769 p += 6;
770 }
771 }
772
773 detail::skip_spaces(p, end);
774 if (p != end) {
775 return false;
776 }
777
778 return is_valid_date_time(dt);
779 }

1105 inline bool sec_of_day(const std::string& str, T& sec) {
1106 if (str.empty()) return false;
1107
1108 const char* p = str.c_str();
1109 int parts[3] = {0, 0, 0}; // hour, minute, second
1110 int idx = 0;
1111
1112 while (*p && idx < 3) {
1113 // Parse integer
1114 int value = 0;
1115 bool has_digit = false;
1116
1117 while (*p >= '0' && *p <= '9') {
1118 has_digit = true;
1119 value = value * 10 + (*p - '0');
1120 ++p;
1121 }
1122
1123 if (!has_digit) return false;
1124 parts[idx++] = value;
1125
1126 // Expect colon or end
1127 if (*p == ':') {
1128 ++p;
1129 } else if (*p == '\0') {
1130 break;
1131 } else {
1132 return false; // unexpected character
1133 }
1134 }
1135
1136 if (idx == 0) return false;
1137 if (!is_valid_time(parts[0], parts[1], parts[2])) return false;
1138
1139 sec = static_cast<T>(sec_of_day(parts[0], parts[1], parts[2]));
1140 return true;
1141 }
